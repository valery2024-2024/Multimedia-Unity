using UnityEngine;

public class ButtonSound : MonoBehaviour
{
    private AudioSource audioSource;   // посилання на AudioSource

    private void Awake()
    {
        // При запуску сцени шукаємо компонент AudioSource
        audioSource = GetComponent<AudioSource>();
    }

    public void PlayClick()
    {
        // Відтворюємо звук натискання
        audioSource.Play();
    }
}
